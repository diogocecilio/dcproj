440  // Row operations for i = 0, ,,,, n - 2 (n-1 not needed)
441  for (int i = 0; i < n - 1; i++)

443  // Partial pivot: find row r below with largest element in column i
444  int r = i;

446  for (int k = i + 1; k < n; k++)
447  {
448  double val = abs(A[k][i]);

1468  const Int n;

1470  VecDoub d;

1473 
1474  Jacobi(MatDoub_I &aa) : n(aa.nrows()), a(aa), v(n, n), d(n), nrot(0),
1475  EPS(numeric_limits<Doub>::epsilon())
1476  {
1477  Int i, j, ip, iq;
1478  Doub tresh, theta, tau, t, sm, s, h, g, c;
1479  VecDoub b(n), z(n);
1480  for (ip = 0;ip < n;ip++) {
1481  for (iq = 0;iq < n;iq++) v[ip][iq] = 0.0;
1482  v[ip][ip] = 1.0;
1483  }
1484  for (ip = 0;ip < n;ip++) {
1485  b[ip] = d[ip] = a[ip][ip];
1486  z[ip] = 0.0;
1487  }
1488  for (i = 1;i <= 50;i++) {
1489  sm = 0.0;
1490  for (ip = 0;ip < n - 1;ip++) {
1491  for (iq = ip + 1;iq < n;iq++)
1492  sm += abs(a[ip][iq]);
1493  }
1494  if (sm == 0.0) {
1495  eigsrt(d, &v);
1496  return;
1497  }
1498  if (i < 4)
1499  tresh = 0.2*sm / (n*n);
1500  else
1501  tresh = 0.0;
1502  for (ip = 0;ip<n - 1;ip++) {
1503  for (iq = ip + 1;iq < n;iq++) {
1504  g = 100.0*abs(a[ip][iq]);
1505  if (i > 4 && g <= EPS*abs(d[ip]) && g <= EPS*abs(d[iq]))
1506  a[ip][iq] = 0.0;
1507  else if (abs(a[ip][iq]) > tresh) {
1508  h = d[iq] - d[ip];
1509  if (g <= EPS*abs(h))
1510  t = (a[ip][iq]) / h;
1511  else {
1512  theta = 0.5*h / (a[ip][iq]);
1513  t = 1.0 / (abs(theta) + sqrt(1.0 + theta*theta));
1514  if (theta < 0.0) t = -t;
1515  }
1516  c = 1.0 / sqrt(1 + t*t);
1517  s = t*c;
1518  tau = s / (1.0 + c);
1519  h = t*a[ip][iq];
1520  z[ip] -= h;
1521  z[iq] += h;
1522  d[ip] -= h;
1523  d[iq] += h;
1524  a[ip][iq] = 0.0;
1525  for (j = 0;j < ip;j++)
1526  rot(a, s, tau, j, ip, j, iq);
1527  for (j = ip + 1;j < iq;j++)
1528  rot(a, s, tau, ip, j, j, iq);
1529  for (j = iq + 1;j < n;j++)
1530  rot(a, s, tau, ip, j, iq, j);
1531  for (j = 0;j < n;j++)
1532  rot(v, s, tau, j, ip, j, iq);
1533  ++nrot;
1534  }
1535  }
1536  }
1537  for (ip = 0;ip < n;ip++) {
1538  b[ip] += z[ip];
1539  d[ip] = b[ip];
1540  z[ip] = 0.0;
1541  }
1542  }
1543  //throw("Too many iterations in routine jacobi");
1544  //std::cout << "Too many iterations in routine jacobi! Critical error!" << std::endl;
1545  fail = true;
1546  }
